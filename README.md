#### Design

Overview: Functions that take basic data type/structs/enum as input

Rust translation:
 - Each type will be a type in rust
 - Pack and Unpack trait that all will implement

Example
-------

STRUCT CALC
int operand1 
int operand2
string operator
ENDSTRUCT

-->

trait RPC {
    fn pack(&self) -> String;
    fn unpack(&String) -> &self;
}

struct CALC {
    operand1: int,
    operant2: int,
    operator: string,
}

impl RPC for CALC {

    fn pack(&self) -> String {
        return in string
    }

    fn unpack(&self) -> Self {
        return in struct
    }
}
